 5 /**
 6  * Un banco aplica un descuento a sus clientes en el pago de membresía por 
 7  * el uso de algunas de sus tarjetas de crédito. Ingresar el tipo de 
 8  * tarjeta y calcular dicho descuento en soles.
 9  * 
10  * |        **Tarjeta**       | **Membresia (S/.)** | **Descuento** |
11  * |:------------------------:|:-------------------:|:-------------:|
12  * | LifeMiles                |              400.00 |            9% |
13  * | Platinium                |              350.00 |            8% |
14  * | Signature                |              300.00 |            7% |
15  * | Oro                      |              190.00 |            6% |
16  * | Repsol, Primera, MCBlack |               75.00 |            5% |
17  * | Otros                    |                0.00 |            0% |
18  *
19  * @author Magh
20  */
21 public class Propuesto04 {
22     public static void main(String[] args) {
23         String tipoTarjeta;
24         double membresia, descPorcent, descMembresia;
25         
26         Scanner teclado = new Scanner(System.in);
27         
28         System.out.println("Ingrese el tipo de tarjeta: ");

32         switch (tipoTarjeta.toLowerCase()) {
33             case "lifemiles":
34                 membresia = 400.00; descPorcent = 0.09;

36             case "platinium":
37                 membresia = 350.00; descPorcent = 0.08;
38                 break;
39             case "signature":
40                 membresia = 300.00; descPorcent = 0.07;
41                 break;
42             case "oro":
43                 membresia = 190.00; descPorcent = 0.06;
44                 break;
45             case "repsol":
46             case "primera":
47             case "mcblack":
48                 membresia = 75.00; descPorcent = 0.05;
49                 break;
50             default:
51                 membresia = 0.00; descPorcent = 0.00;
52                 break;
53         }
54         
55         descMembresia = membresia * descPorcent;
56         System.out.printf("El descuento en la membresía de la "
57         + "tarjeta %s es: S/ %.2f\n", tipoTarjeta, descMembresia);
58         
59         teclado.close();
60     }
